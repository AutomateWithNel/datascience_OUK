# SQL query to join customers and orders and retrieve information
 
query = '''
 
    
SELECT customers.name, customers.country, orders.order_id, orders.order_date, 
orders.amount 
 
    
FROM customers
 
    
JOIN orders ON 
customers.customer_id = orders.customer_id
 
'''
 
 
# Execute the query
 
cursor.execute(query)
 
results = cursor.fetchall()
 
 
# Display results
 
for row in results:
 
    
print(row)
